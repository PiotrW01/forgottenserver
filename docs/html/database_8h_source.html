183class DBTransaction

198
199 bool begin()
200 {
201 state = STATE_START;
202 return Database::getInstance().beginTransaction();
203 }
204
205 bool commit()
206 {
207 if (state != STATE_START) {
208 return false;
209 }
210
211 state = STATE_COMMIT;
212 return Database::getInstance().commit();
213 }
214
215private:

218 STATE_NO_START,
