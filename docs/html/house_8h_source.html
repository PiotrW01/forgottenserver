223 Houses(const Houses&) = delete;
224 Houses& operator=(const Houses&) = delete;
225
226 House* addHouse(uint32_t id)
227 {
228 auto it = houseMap.find(id);
229 if (it != houseMap.end()) {
230 return it->second;
231 }
232
233 House* house = new House(id);
234 houseMap[id] = house;
235 return house;
236 }
237
238 House* getHouse(uint32_t houseId)
239 {
240 auto it = houseMap.find(houseId);
241 if (it == houseMap.end()) {
242 return nullptr;
243 }
244 return it->second;
245 }
