99 /* Load the spawns
100 * \param map pointer to the Map class
101 * \returns Returns true if the spawns were loaded successfully
102 */
103 static bool loadSpawns(Map* map, bool isCalledByLua)
104 {
105 if (map->spawnfile.empty()) {
106 // OTBM file doesn't tell us about the spawnfile, lets guess it is mapname-spawn.xml.
107 map->spawnfile = getString(ConfigManager::MAP_NAME);
108 map->spawnfile += "-spawn.xml";
109 }
110
111 return map->spawns.loadFromXml(map->spawnfile.string(), isCalledByLua);
112 }
113
114 /* Load the houses (not house tile-data)
115 * \param map pointer to the Map class
116 * \returns Returns true if the houses were loaded successfully
117 */
118 static bool loadHouses(Map* map)
119 {
120 if (map->housefile.empty()) {
121 // OTBM file doesn't tell us about the housefile, lets guess it is mapname-house.xml.
122 map->housefile = getString(ConfigManager::MAP_NAME);
123 map->housefile += "-house.xml";
124 }
125
126 return map->houses.loadHousesXML(map->housefile.string());
127 }

129 const std::string& getLastErrorString() const { return errorString; }

131 void setLastErrorString(std::string error) { errorString = error; }

136 bool parseWaypoints(OTB::Loader& loader, const OTB::Node& waypointsNode, Map& map);
