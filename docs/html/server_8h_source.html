96template <typename ProtocolType>
97bool ServiceManager::add(uint16_t port)
98{
99 if (port == 0) {
100 std::cout << "ERROR: No port provided for service " << ProtocolType::protocol_name() << ". Service disabled."
101 << std::endl;
102 return false;
103 }
104
105 ServicePort_ptr service_port;
106
107 auto foundServicePort = acceptors.find(port);
108
109 if (foundServicePort == acceptors.end()) {
110 service_port = std::make_shared<ServicePort>(io_context);
111 service_port->open(port);
112 acceptors[port] = service_port;
113 } else {
114 service_port = foundServicePort->second;
115
116 if (service_port->is_single_socket() || ProtocolType::server_sends_first) {
117 std::cout << "ERROR: " << ProtocolType::protocol_name() << " and " << service_port->get_protocol_names()
118 << " cannot use the same port " << port << '.' << std::endl;
119 return false;
120 }
121 }
122
123 return service_port->add_service(std::make_shared<Service<ProtocolType>>());
124}
