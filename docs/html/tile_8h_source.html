253// Used for walkable tiles, where there is high likeliness of
254// items being added/removed
255class DynamicTile : public Tile
256{
257 // By allocating the vectors in-house, we avoid some memory fragmentation
258 TileItemVector items;
259 CreatureVector creatures;
260
261public:
262 DynamicTile(uint16_t x, uint16_t y, uint8_t z) : Tile(x, y, z) {}
263 ~DynamicTile()
264 {
265 for (Item* item : items) {
266 item->decrementReferenceCounter();
267 }
268 }
269
270 using Tile::internalAddThing;
271
272 // non-copyable
273 DynamicTile(const DynamicTile&) = delete;
274 DynamicTile& operator=(const DynamicTile&) = delete;
275
276 TileItemVector* getItemList() override { return &items; }
277 const TileItemVector* getItemList() const override { return &items; }
278 TileItemVector* makeItemList() override { return &items; }
279
280 CreatureVector* getCreatures() override { return &creatures; }
281 const CreatureVector* getCreatures() const override { return &creatures; }
282 CreatureVector* makeCreatures() override { return &creatures; }

285// For blocking tiles, where we very rarely actually have items
286class StaticTile final : public Tile
287{
288 // We very rarely even need the vectors, so don't keep them in memory
289 std::unique_ptr<TileItemVector> items;
290 std::unique_ptr<CreatureVector> creatures;
291
292public:
293 StaticTile(uint16_t x, uint16_t y, uint8_t z) : Tile(x, y, z) {}
294 ~StaticTile()
295 {
296 if (items) {
297 for (Item* item : *items) {
298 item->decrementReferenceCounter();
299 }
300 }
301 }
302
303 using Tile::internalAddThing;
304
305 // non-copyable
306 StaticTile(const StaticTile&) = delete;
307 StaticTile& operator=(const StaticTile&) = delete;
308
309 TileItemVector* getItemList() override { return items.get(); }
310 const TileItemVector* getItemList() const override { return items.get(); }
311 TileItemVector* makeItemList() override
312 {
313 if (!items) {
314 items.reset(new TileItemVector);
315 }
316 return items.get();
317 }
318
319 CreatureVector* getCreatures() override { return creatures.get(); }
320 const CreatureVector* getCreatures() const override { return creatures.get(); }
321 CreatureVector* makeCreatures() override
322 {
323 if (!creatures) {
324 creatures.reset(new CreatureVector);
325 }
326 return creatures.get();
327 }
328};
