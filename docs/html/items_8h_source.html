466 void emplace(uint16_t clientId, uint16_t serverId)
467 {
468 if (clientId >= vec.size()) {
469 vec.resize(clientId + 1, 0);

471 if (vec[clientId] == 0) {
472 vec[clientId] = serverId;
473 }
474 }

477 {
478 uint16_t serverId = 0;

481 }
482 return serverId;
483 }
