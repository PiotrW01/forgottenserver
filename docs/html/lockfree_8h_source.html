12/*
13 * we use this to avoid instantiating multiple free lists for objects of the
14 * same size and it can be replaced by a variable template in C++14
15 *
16 * template <size_t TSize, size_t Capacity>
17 * boost::lockfree::stack<void*, boost::lockfree::capacity<Capacity>
18 * lockfreeFreeList;
19 */
20template <size_t TSize, size_t Capacity>
21struct LockfreeFreeList
22{
23 using FreeList = boost::lockfree::stack<void*, boost::lockfree::capacity<Capacity>>;
24 static FreeList& get()
25 {
26 static FreeList freeList;
27 return freeList;
28 }
29};

31template <typename T, size_t Capacity>
32class LockfreePoolingAllocator
33{
34public:
35 template <class U>
36 struct rebind
37 {
38 using other = LockfreePoolingAllocator<U, Capacity>;
39 };
40
41 LockfreePoolingAllocator() = default;
42
43 template <typename U>
44 explicit constexpr LockfreePoolingAllocator(const LockfreePoolingAllocator<U, Capacity>&)
45 {}
46 using value_type = T;
47
48 T* allocate(size_t) const
49 {
50 auto& inst = LockfreeFreeList<sizeof(T), Capacity>::get();
51 void* p; // NOTE: p doesn't have to be initialized
52 if (!inst.pop(p)) {
53 // Acquire memory without calling the constructor of T
54 p = operator new(sizeof(T));
55 }
56 return static_cast<T*>(p);
57 }
58
59 void deallocate(T* p, size_t) const
60 {
61 auto& inst = LockfreeFreeList<sizeof(T), Capacity>::get();
62 if (!inst.bounded_push(p)) {
63 // Release memory without calling the destructor of T
64 //(it has already been called at this point)
65 operator delete(p);
66 }
67 }
68};
